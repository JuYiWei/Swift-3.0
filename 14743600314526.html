简介

Protocols 官方文档

一个协议定义了适合特定任务或一项功能的方法，属性和其他要求的蓝图。该协议可以被采用由一个类，结构，或枚举，以提供一个实际实现的这些要求。满足协议要求的任何类型被认为符合该协议。

除了指定符合类型必须实现的要求，可以延长一个协议来实现其中的一些要求，或实施符合类型可以利用的附加功能。

协议语法

  • 一般协议
protocol SomeProtocol {
    // protocol definition goes here
}
  • 结构体，采用多个协议
struct SomeStructure: FirstProtocol, AnotherProtocol {
    // structure definition goes here
}
  • 类继承父类，又采用多个协议
class SomeClass: SomeSuperclass, FirstProtocol, AnotherProtocol {
    // class definition goes here
}

协议添加属性

  • 协议添加属性要求。
protocol FullName {
    var fullName: String { get }
}

class Person : FullName{
    var firstName = ""
    var lastName = ""
    init(firstName: String, lastName: String) {
        self.firstName = firstName
        self.lastName = lastName
    }
    
    var fullName: String  {
        return firstName + lastName
    }
}

var p = Person(firstName: "zhang", lastName: "san")
p.fullName

协议添加方法

protocol RandomTest {
    func random() -> Double
}

class TestClass : RandomTest {
    var a: Double
    init(a: Double) {
        self.a = a
    }
    
    func random() -> Double {
        return a * a
    }
}

var t = TestClass(a: 123)
t.random()

协议添加突变

protocol Togglable {
    mutating func toggle()
}

enum OnOffSwitch: Togglable {
    case off, on
    
    mutating func toggle() {
        switch self {
        case .off:
            self = .on
        case .on:
            self = .off
        }
    }
}

var lightSwitch = OnOffSwitch.off
lightSwitch.toggle()

协议添加初始化

protocol MyInit {
    init(a: String, b: String)
}

指定初始化

  • 一般协议指定初始化
protocol MyInit {
    init(a: String, b: String)
}

class MyClass: MyInit {
    required init(a: String, b: String) {
        print(a+b)
    }
}
  • 子类重写并且协议指定初始化
protocol MyInit {
    init(a: String, b: String)
}

class MyClass {
    init(a: String, b: String) {
        print(a+b)
    }
}

class SubClass: MyClass, MyInit {
    required override init(a: String, b:
        String) {
        super.init(a: a, b: b)
        print(a + a + b)
    }
}

Failable初始化

  • 与上述类似（略）。

作为协议的类型

  • 协议本身，也可以看成是类。
protocol TestDelegate {
    func complete(name: String)
}

class DelegateClass: TestDelegate {
    func complete(name: String) {
        print(name)
    }
}

委托

  • 与OC的委托代理有一点不同，但又很相似。
protocol TestDelegate {
    func complete(name: String)
}

class TestClass {
    var delegate: TestDelegate?       
    func play() {
        delegate?.complete(name: "123")
    }
}

class DelegateClass: TestDelegate {
    func complete(name: String) {
        print(name)
    }
}

var d = DelegateClass()
var c = TestClass()
c.delegate = d
c.play()

扩展添加协议

protocol TestDelegate {
    func complete(log: String)
}

extension String: TestDelegate {

        print(log)
    }
}

已符合协议内容的类，添加声明协议的拓展

protocol TestDelegate {
    func complete(log: String)
}

class MyString {
    func complete(log: String) {
        print(log)
    }
}

extension MyString: TestDelegate { }

var s = MyString()
s.complete(log: "???")

协议类型的集合

  • 由于协议本身也是类，所以可以存储在集合中。
protocol TestDelegate {
    func complete(log: String)
}

class DelegateClass:TestDelegate {
    func complete(log: String) {
        print(log)
    }
}

var arr = [DelegateClass(), DelegateClass()]
arr[0].complete(log: "1234")

协议继承

  • 协议也是类，协议继承协议，包括了原有的协议内容，也可以继承多个协议，用逗号分隔。
protocol TestProtocol {
    func complete(log: String)
}

protocol TestProtocol2: TestProtocol {
    func stop(desc: String)
}

class TestClass: TestProtocol2 {
    
    func complete(log: String) {
        print(log)
    }
    
    func stop(desc: String) {
        print(desc)
    }
}

var a = TestClass()
a.complete(log: "log")
a.stop(desc: "stop")

只允许类使用的协议

protocol TestProtocol {
    func complete(log: String)
}

protocol OtherProtocol: class, TestProtocol {
    // 只有class才能使用的协议
}

struct TestEnum: OtherProtocol {
    func complete(log: String) {
        print(123)
    }
}

协议组合与分解

protocol NamaProtocol {
    var name: String { get }
}

protocol AgeProtocol {
    var age: Int { get }
}

struct Person: NamaProtocol, AgeProtocol{
    var name: String
    var age:Int
}

func welcome(wel:NamaProtocol & AgeProtocol) {
    print("\(wel.name) \(wel.age)")
}

var per = Person(name: "涨三", age: 12)
welcome(wel: per)

检查协议一致性

  • 准备
protocol NamaProtocol {
    var name: String { get }
}
protocol AgeProtocol {
    var age: Int { get }
}

class A: NamaProtocol {
    var name: String = ""
}
class B: AgeProtocol {
    var age: Int = 12
}
  • 直接判断
var a = A()
var b = B()

if a is NamaProtocol {
    print(1)
}
  • 可选类型判断，关于可选类型略。
if let x = a as? NamaProtocol {
    print(1)
}

或者

let x = a as? NamaProtocol
let y = a as! NamaProtocol

可选协议

  • 注意可选协议的语法，以及可选协议的判断，如同可选类型一样。
@objc protocol NamaProtocol {
    @objc optional func increase()
}
  • 简单举例，与一般协议使用类似。
@objc protocol NamaProtocol {
    @objc optional func increase()
}

class ClassA {
    var delegate: NamaProtocol?
    
    func increase() {
        delegate?.increase!()
    }
}

class Other: NamaProtocol {
    func increase() {
        print(111)
    }
}


var classA = ClassA()
classA.delegate = Other()
for index in 1...4 {
    classA.increase()
    print(index)
}

提供了默认的实现

protocol NamaProtocol {
    func increase()
}

class TestDelegate:NamaProtocol {
    func increase() {
        print("i")
    }
    
    func reduce() {
        print("??")
    }// 相对于自定义实现，如果没有则使用默认实现。
}

extension NamaProtocol {
    func reduce() {
        print("r")
    }// 默认实现。
}

var A = TestDelegate()
A.increase()
A.reduce()

选择性添加拓展？并不理解

protocol TextRepresentable {
    var textualDescription: String { get }
}

struct Hamster: TextRepresentable {
    var name: String
    var textualDescription: String {
        return "A hamster named \(name)"
    }
}


extension Collection where Iterator.Element: TextRepresentable {
    var textualDescription: String {
        let itemsAsText = self.map { $0.textualDescription }
        return "[" + itemsAsText.joined(separator: ", ") + "]"
    }
}


let murrayTheHamster = Hamster(name: "Murray")
let morganTheHamster = Hamster(name: "Morgan")
let mauriceTheHamster = Hamster(name: "Maurice")
let hamsters = [murrayTheHamster, morganTheHamster, mauriceTheHamster]

print(hamsters.textualDescription)
